

 

procedure insertionSort(var arr[]: array of integer, var n:integer)

   var i, key, j : integer;
   n=arr.length
    for (i = 0; i < n-1; i++)
    
        key = arr[i];
        j = i - 1;
 
        while (j >= 0 && arr[j] > key)
        
            arr[j + 1] = arr[j];
            j = j - 1;
        
        arr[j + 1] = key;
